```tsx
"use client";

import React, { useEffect, useState } from "react";

type IntroGateProps = {
  children: React.ReactNode;
};

export function IntroGate({ children }: IntroGateProps) {
  const [showIntro, setShowIntro] = useState(true);
  const [hasStarted, setHasStarted] = useState(false);

  // Safety timeout: hide intro even if video doesn't autoplay.
  useEffect(() => {
    if (!showIntro) return;
    const timeout = setTimeout(() => {
      setShowIntro(false);
    }, 8000); // 8 seconds fallback
    return () => clearTimeout(timeout);
  }, [showIntro]);

  const handleVideoCanPlay = () => {
    setHasStarted(true);
  };

  const handleVideoEnd = () => {
    setShowIntro(false);
  };

  const handleSkip = () => {
    setShowIntro(false);
  };

  return (
    <>
      {showIntro && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-slate-950">
          <div className="absolute inset-0 pointer-events-none bg-slate-950/40" />
          <div className="relative z-10 flex h-full w-full flex-col items-center justify-center">
            <div className="relative w-full max-w-4xl px-4">
              <div className="overflow-hidden rounded-3xl border border-slate-800 bg-black shadow-2xl">
                <video
                  src="/intro.mp4"
                  className="h-[60vh] w-full object-cover sm:h-[70vh]"
                  autoPlay
                  muted
                  playsInline
                  onCanPlay={handleVideoCanPlay}
                  onEnded={handleVideoEnd}
                />
              </div>

              <div className="mt-4 flex items-center justify-between text-xs text-slate-300 sm:text-sm">
                <span>
                  {hasStarted
                    ? "SuVerse Labs · AI · Blockchain · Green Logistics"
                    : "Loading intro…"}
                </span>
                <button
                  type="button"
                  onClick={handleSkip}
                  className="inline-flex items-center gap-1 rounded-full border border-slate-600 bg-slate-900/80 px-4 py-1.5 text-xs font-medium text-slate-100 shadow-sm transition hover:border-slate-300 hover:bg-slate-800/90"
                >
                  Skip intro
                </button>
              </div>
            </div>
          </div>
        </div>
      )}

      {/* Main app content */}
      <div className={showIntro ? "opacity-0" : "opacity-100 transition-opacity duration-500"}>
        {children}
      </div>
    </>
  );
}

Notes:
	•	showIntro controls whether the overlay is visible.
	•	IntroGate always renders children; they’re just faded out while showIntro is true.
	•	The video file path is /intro.mp4, which comes from public/intro.mp4.
